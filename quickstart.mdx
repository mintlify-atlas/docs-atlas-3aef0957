## React quickstart

Use `@videojs/react` to build players with composable React components and hooks.

<Steps>

<Step title="Install the package">

<CodeGroup>

```bash npm
npm install @videojs/react
```

```bash pnpm
pnpm add @videojs/react
```

```bash yarn
yarn add @videojs/react
```

```bash bun
bun add @videojs/react
```

</CodeGroup>

</Step>

<Step title="Create your player component">

Create a new file for your player. Call `createPlayer` with a feature bundle to get a typed `Player` object that contains the `Provider`, `Container`, and a `usePlayer` hook scoped to that player instance.

```tsx title="components/player.tsx"
'use client';

import '@videojs/react/video/skin.css';
import { createPlayer } from '@videojs/react';
import { Video, VideoSkin, videoFeatures } from '@videojs/react/video';

const Player = createPlayer({ features: videoFeatures });

interface MyPlayerProps {
  src: string;
}

export const MyPlayer = ({ src }: MyPlayerProps) => {
  return (
    <Player.Provider>
      <VideoSkin>
        <Video src={src} playsInline />
      </VideoSkin>
    </Player.Provider>
  );
};
```

<Note>
The `'use client'` directive is required for Next.js App Router. Omit it for Vite, Remix, or other setups that don't use React Server Components.
</Note>

</Step>

<Step title="Use the player in your app">

Import and render your player component anywhere in your app:

```tsx title="app/page.tsx"
import { MyPlayer } from '../components/player';

export default function Page() {
  return (
    <MyPlayer src="https://stream.mux.com/BV3YZtogl89mg9VcNBhhnHm02Y34zI1nlMuMQfAbl3dM/highest.mp4" />
  );
}
```

</Step>

<Step title="Play an HLS stream (optional)">

To stream with HLS, swap `Video` for `HlsVideo` from the media subpath. The feature bundle and skin stay the same:

```tsx title="components/player.tsx"
'use client';

import '@videojs/react/video/skin.css';
import { createPlayer } from '@videojs/react';
import { VideoSkin, videoFeatures } from '@videojs/react/video';
import { HlsVideo } from '@videojs/react/media/hls-video';

const Player = createPlayer({ features: videoFeatures });

export const MyPlayer = ({ src }: { src: string }) => {
  return (
    <Player.Provider>
      <VideoSkin>
        <HlsVideo src={src} playsInline />
      </VideoSkin>
    </Player.Provider>
  );
};
```

</Step>

</Steps>

## HTML quickstart

Use `@videojs/html` to build players with framework-free custom elements that work in any web environment.

<Tabs>
  <Tab title="npm / module bundler">

<Steps>

<Step title="Install the package">

<CodeGroup>

```bash npm
npm install @videojs/html
```

```bash pnpm
pnpm add @videojs/html
```

```bash yarn
yarn add @videojs/html
```

```bash bun
bun add @videojs/html
```

</CodeGroup>

</Step>

<Step title="Register the custom elements">

Import the side-effect modules that register the custom elements, and import the skin CSS:

```javascript title="index.js"
import '@videojs/html/video/player';
import '@videojs/html/video/skin';
import '@videojs/html/video/skin.css';
```

Each import registers one custom element. The `/video/player` module registers `<video-player>`, `/video/skin` registers `<video-skin>`.

</Step>

<Step title="Add the player markup">

```html title="index.html"
<video-player>
  <video-skin>
    <video slot="media" src="https://stream.mux.com/BV3YZtogl89mg9VcNBhhnHm02Y34zI1nlMuMQfAbl3dM/highest.mp4" playsinline></video>
  </video-skin>
</video-player>
```

The `slot="media"` attribute tells the player which element is the media source. Always include it on your `<video>`, `<audio>`, or custom media element.

</Step>

<Step title="Play an HLS stream (optional)">

To use adaptive HLS streaming, add the `hls-video` import and swap the media element:

```javascript title="index.js"
import '@videojs/html/video/player';
import '@videojs/html/video/skin';
import '@videojs/html/video/skin.css';
import '@videojs/html/media/hls-video';
```

```html title="index.html"
<video-player>
  <video-skin>
    <hls-video slot="media" src="https://stream.mux.com/BV3YZtogl89mg9VcNBhhnHm02Y34zI1nlMuMQfAbl3dM.m3u8" playsinline></hls-video>
  </video-skin>
</video-player>
```

</Step>

</Steps>

  </Tab>
  <Tab title="CDN">

<Steps>

<Step title="Add the script and stylesheet">

Load the player and skin from the CDN. No build step required:

```html title="index.html"
<!doctype html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/@videojs/html/cdn/video/skin.css" />
</head>
<body>

  <video-player>
    <video-skin>
      <video slot="media" src="https://stream.mux.com/BV3YZtogl89mg9VcNBhhnHm02Y34zI1nlMuMQfAbl3dM/highest.mp4" playsinline></video>
    </video-skin>
  </video-player>

  <script type="module" src="https://cdn.jsdelivr.net/npm/@videojs/html/cdn/video/player.js"></script>

</body>
</html>
```

</Step>

<Step title="Play an HLS stream (optional)">

Add the HLS script to enable adaptive streaming:

```html title="index.html"
<script type="module" src="https://cdn.jsdelivr.net/npm/@videojs/html/cdn/video/player.js"></script>
<script type="module" src="https://cdn.jsdelivr.net/npm/@videojs/html/cdn/media/hls-video.js"></script>
```

```html
<video-player>
  <video-skin>
    <hls-video slot="media" src="https://stream.mux.com/BV3YZtogl89mg9VcNBhhnHm02Y34zI1nlMuMQfAbl3dM.m3u8" playsinline></hls-video>
  </video-skin>
</video-player>
```

</Step>

</Steps>
